2. ruta u web.php

// Views and passing data -----------------------------
// nova home ruta
// http://localhost:8000
Route::get('/', function () {
    // passing data as array
    // vrati view listings.php i pošalji mu data
    return view('listings', [
        'heading' => 'Latest Listings'
    ]);
});
